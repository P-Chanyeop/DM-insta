### 자주 쓰는 API 호출 (preview_eval에 복붙)
```js
// 토큰
localStorage.getItem('authToken')

// 자동화 목록
const tok = localStorage.getItem('authToken')
fetch('http://localhost:8080/api/automations', { headers: { Authorization: `Bearer ${tok}` }}).then(r=>r.json())

// 플로우 목록
fetch('http://localhost:8080/api/flows', { headers: { Authorization: `Bearer ${tok}` }}).then(r=>r.json())

// billing info (S5 버그 재확인)
fetch('http://localhost:8080/api/billing/info', { headers: { Authorization: `Bearer ${tok}` }}).then(r=>r.json())

// 트리거 quota 강제 호출 (S5 백엔드 검증 OK)
fetch('http://localhost:8080/api/automations', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${tok}` },
  body: JSON.stringify({ name: '테스트', type: 'ICEBREAKER', responseMessage: 't' })
}).then(r=>r.text())
```

### 빠른 신규 가입 (이메일 인증 우회)
```js
fetch('http://localhost:8080/api/auth/signup', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ email: 'qa@test.com', password: 'Test1234!', name: '검증유저' })
}).then(r=>r.json())

// 그 다음 로그인 (emailVerified 우회 가능)
fetch('http://localhost:8080/api/auth/login', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ email: 'qa@test.com', password: 'Test1234!' })
}).then(r=>r.json()).then(d=>{
  localStorage.setItem('authToken', d.token)
  localStorage.setItem('authUser', JSON.stringify(d.user || {}))
  // 온보딩 스킵 마크
  if (d.user?.email) localStorage.setItem(`onboarding_completed_${d.user.email}`, 'true')
})
```
